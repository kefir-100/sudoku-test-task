# Sudoku Angular 21 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a production-ready Sudoku web app in Angular 21 (zoneless, Signals, standalone) consuming the Sugoku API, deployable from a single `npm install && npm run serve:local`.

**Architecture:** Layered: Components (OnPush, signal inputs) → `SudokuStore` (signal store) → `SudokuApiService` + `BoardMappingService` + `GameStateStorageService` + `ErrorHandlerService`. HTTP interceptor encodes Sugoku's required `application/x-www-form-urlencoded` bodies. Storage uses an abstract-class DI token (`StorageProvider`) with `LocalStorageService` as the default implementation. Routing: `/` (Welcome) + lazy-loaded `/play` (Game).

**Tech Stack:** Angular 21 (zoneless by default), Angular Material 21 + CDK, Tailwind CSS v4 (CSS-first config), TypeScript strict, Vitest via `@angular/build:unit-test`, ESLint flat config + `@angular-eslint` + `typescript-eslint`, Prettier, Stylelint, Husky + lint-staged.

**Spec:** [`2026-05-20-sudoku-angular-design.md`](./2026-05-20-sudoku-angular-design.md) (immutable).
**Progress log:** [`2026-05-20-sudoku-angular-progress.md`](./2026-05-20-sudoku-angular-progress.md) (living).

import type { RawBoard } from '../../models/sudoku-api-response';

export const encodeBoardParam = (board: RawBoard): string =>
  board.reduce(
    (accumulator, row, index) =>
      accumulator +
      `%5B${encodeURIComponent(row.join(','))}%5D${index === board.length - 1 ? '' : '%2C'}`,
    '',
  );

export const encodeFormUrlEncoded = (params: Record<string, RawBoard>): string =>
  Object.keys(params)
    .map((key) => {
      const board = params[key];
      if (!board) {
        return '';
      }
      return `${key}=%5B${encodeBoardParam(board)}%5D`;
    })
    .filter((part) => part.length > 0)
    .join('&');
```

import type { HttpInterceptorFn } from '@angular/common/http';
import { APP_CONFIG } from '../config/app-config';
import { encodeFormUrlEncoded } from './encode-board.helper';
import type { RawBoard } from '../../models/sudoku-api-response';

export const formUrlEncodedInterceptor: HttpInterceptorFn = (req, next) => {
  if (!req.url.startsWith(APP_CONFIG.api.baseUrl)) {
    return next(req);
  }
  if (req.method !== 'POST') {
    return next(req);
  }
  const body = req.body as Record<string, RawBoard> | null;
  const encodedBody = body ? encodeFormUrlEncoded(body) : null;
  const formReq = req.clone({
    body: encodedBody,
    setHeaders: { 'Content-Type': 'application/x-www-form-urlencoded' },
  });
  return next(formReq);
};
```

import { HttpClient, HttpParams } from '@angular/common/http';
import { inject, Injectable } from '@angular/core';
import type { Observable } from 'rxjs';
import { APP_CONFIG } from '../config/app-config';
import type { GameDifficultyLevel } from '../../models/game-difficulty-level';
import type {
  BoardApiResponse,
  RawBoard,
  SolveApiResponse,
  ValidateApiResponse,
} from '../../models/sudoku-api-response';

export class SudokuApiService {

    return this.http.post<ValidateApiResponse>(
      `${APP_CONFIG.api.baseUrl}${APP_CONFIG.api.endpoints.validate}`,
      { board },
    );
  }
}

import { Injectable } from '@angular/core';
import type { RawBoard } from '../../models/sudoku-api-response';
import type { SudokuBoard, SudokuCell } from '../../models/sudoku-board';

const SIZE = 9;

@Injectable({ providedIn: 'root' })
export class BoardMappingService {
  toUiBoard(rawBoard: RawBoard): SudokuBoard {
    const cells: SudokuCell[] = [];
    for (let row = 0; row < SIZE; row++) {
      for (let column = 0; column < SIZE; column++) {
        const value = rawBoard[row]?.[column] ?? 0;
        cells.push({
          id: row * SIZE + column,
          row,
          column,
          boxIndex: Math.floor(row / 3) * 3 + Math.floor(column / 3),
          originalValue: value,
          userValue: value,
          isPrefilled: value !== 0,
          isSolverFilled: false,
        });
      }
    }
    return { cells };
  }

  toWire(board: SudokuBoard): RawBoard {
    const rows: number[][] = Array.from({ length: SIZE }, () => Array<number>(SIZE).fill(0));
    for (const cell of board.cells) {
      const targetRow = rows[cell.row];
      if (targetRow) {
        targetRow[cell.column] = cell.userValue;
      }
    }
    return rows;
  }

  updateUserValue(board: SudokuBoard, cellId: number, value: number): SudokuBoard {
    return {
      cells: board.cells.map((cell) =>
        cell.id === cellId && !cell.isPrefilled
          ? { ...cell, userValue: value, isSolverFilled: false }
          : cell,
      ),
    };
  }

  applySolution(board: SudokuBoard, solution: RawBoard): SudokuBoard {
    return {
      cells: board.cells.map((cell) => {
        if (cell.isPrefilled) {
          return cell;
        }
        const solutionValue = solution[cell.row]?.[cell.column] ?? 0;
        const isSolverFilled = cell.userValue !== solutionValue || cell.userValue === 0;
        return { ...cell, userValue: solutionValue, isSolverFilled };
      }),
    };
  }
}

import { HttpErrorResponse } from '@angular/common/http';
import { ErrorHandler, inject, Injectable } from '@angular/core';
import { MatSnackBar } from '@angular/material/snack-bar';
import { APP_CONFIG } from '../config/app-config';
import type { NormalizedError } from '../../models/normalized-error';

@Injectable({ providedIn: 'root' })
export class ErrorHandlerService implements ErrorHandler {
  private readonly snackBar = inject(MatSnackBar);

  handleHttpError(error: HttpErrorResponse): NormalizedError {
    const kind: NormalizedError['kind'] = error.status === 0 ? 'network' : 'server';
    const normalized: NormalizedError = {
      kind,
      message: APP_CONFIG.snackbar.messages.apiError,
      httpStatus: error.status,
    };
    this.notify(normalized.message);
    return normalized;
  }

  handleStorageError(error: unknown): NormalizedError {
    const message = error instanceof Error ? error.message : 'Storage failed.';
    const normalized: NormalizedError = { kind: 'storage', message };
    this.notify(message);
    return normalized;
  }

  handleError(error: unknown): void {
    console.error('[Sudoku] unhandled error', error);
  }

  private notify(message: string): void {
    this.snackBar.open(message, 'Dismiss', {
      duration: APP_CONFIG.snackbar.durationMs,
    });
  }
}

@Injectable({ providedIn: 'root' })

    localStorage.removeItem(key);
  }

  clear(): void {
    localStorage.clear();
  }
}

// src/app/features/game/helpers/conflict-detection.ts
import type { SudokuBoard } from '../../../models/sudoku-board';

export const computeConflictCellIds = (board: SudokuBoard): ReadonlySet<number> => {
  const conflicts = new Set<number>();
  const groups: Record<string, Map<number, number[]>> = {
    row: new Map(),
    column: new Map(),
    box: new Map(),
  };
  for (const cell of board.cells) {
    if (cell.userValue === 0) {
      continue;
    }
    const trackers: Array<[keyof typeof groups, number]> = [
      ['row', cell.row],
      ['column', cell.column],
      ['box', cell.boxIndex],
    ];
    for (const [kind, group] of trackers) {
      const groupMap = groups[kind];
      if (!groupMap) {
        continue;
      }
      const bucket = groupMap.get(group * 10 + cell.userValue) ?? [];
      bucket.push(cell.id);
      groupMap.set(group * 10 + cell.userValue, bucket);
    }
  }
  for (const map of Object.values(groups)) {
    for (const ids of map.values()) {
      if (ids.length > 1) {
        for (const id of ids) {
          conflicts.add(id);
        }
      }
    }
  }
  return conflicts;
};

// src/app/features/game/helpers/highlight-computation.ts
import type { SudokuBoard } from '../../../models/sudoku-board';

export const computeRelatedCellIds = (
  board: SudokuBoard,
  focusedCellId: number | null,
): ReadonlySet<number> => {
  if (focusedCellId === null) {
    return new Set();
  }
  const focused = board.cells.find((c) => c.id === focusedCellId);
  if (!focused) {
    return new Set();
  }
  const related = new Set<number>();
  for (const cell of board.cells) {
    if (cell.id === focused.id) {
      continue;
    }
    if (
      cell.row === focused.row ||
      cell.column === focused.column ||
      cell.boxIndex === focused.boxIndex
    ) {
      related.add(cell.id);
    }
  }
  return related;
};

export const computeSameNumberCellIds = (
  board: SudokuBoard,
  focusedCellId: number | null,
): ReadonlySet<number> => {
  if (focusedCellId === null) {
    return new Set();
  }
  const focused = board.cells.find((c) => c.id === focusedCellId);
  if (!focused || focused.userValue === 0) {
    return new Set();
  }
  const matches = new Set<number>();
  for (const cell of board.cells) {
    if (cell.id !== focused.id && cell.userValue === focused.userValue) {
      matches.add(cell.id);
    }
  }
  return matches;
};

// src/app/features/game/sudoku.store.ts
import { HttpErrorResponse } from '@angular/common/http';
import { computed, DestroyRef, effect, inject, Injectable, signal } from '@angular/core';
import { takeUntilDestroyed } from '@angular/core/rxjs-interop';
import { MatSnackBar } from '@angular/material/snack-bar';
import { catchError, EMPTY, Subject, switchMap, tap } from 'rxjs';
import { APP_CONFIG } from '../../core/config/app-config';
import { SudokuApiService } from '../../core/api/sudoku-api.service';
import { BoardMappingService } from '../../core/mapping/board-mapping.service';
import { ErrorHandlerService } from '../../core/errors/error-handler.service';
import { GameStateStorageService } from '../../core/storage/game-state-storage.service';
import type {
  GameDifficultyLevel,
  ResolvedDifficultyLevel,
} from '../../models/game-difficulty-level';
import type { GameStatus } from '../../models/game-status';
import type { GameSnapshot } from '../../models/game-snapshot';
import type { NormalizedError } from '../../models/normalized-error';
import type { SudokuBoard } from '../../models/sudoku-board';
import type { SolveApiStatus, ValidateApiStatus } from '../../models/sudoku-api-response';
import { computeConflictCellIds } from './helpers/conflict-detection';
import { computeRelatedCellIds, computeSameNumberCellIds } from './helpers/highlight-computation';

@Injectable({ providedIn: 'root' })
export class SudokuStore {
  private readonly api = inject(SudokuApiService);
  private readonly mapper = inject(BoardMappingService);
  private readonly errors = inject(ErrorHandlerService);
  private readonly storage = inject(GameStateStorageService);
  private readonly snackBar = inject(MatSnackBar);
  private readonly destroyRef = inject(DestroyRef);

  private readonly _status = signal<GameStatus>('idle');
  private readonly _difficulty = signal<GameDifficultyLevel | null>(null);
  private readonly _resolvedDifficulty = signal<ResolvedDifficultyLevel | null>(null);
  private readonly _board = signal<SudokuBoard | null>(null);
  private readonly _focusedCellId = signal<number | null>(null);
  private readonly _validationResult = signal<ValidateApiStatus | null>(null);
  private readonly _solveResult = signal<SolveApiStatus | null>(null);
  private readonly _error = signal<NormalizedError | null>(null);

  private readonly validateAction = new Subject<void>();
  private readonly solveAction = new Subject<void>();

  readonly status = this._status.asReadonly();
  readonly difficulty = this._difficulty.asReadonly();
  readonly resolvedDifficulty = this._resolvedDifficulty.asReadonly();
  readonly board = this._board.asReadonly();
  readonly focusedCellId = this._focusedCellId.asReadonly();
  readonly validationResult = this._validationResult.asReadonly();
  readonly solveResult = this._solveResult.asReadonly();
  readonly error = this._error.asReadonly();

  readonly conflictCellIds = computed(() => {
    const board = this._board();
    return board ? computeConflictCellIds(board) : new Set<number>();
  });
  readonly relatedCellIds = computed(() => {
    const board = this._board();
    return board ? computeRelatedCellIds(board, this._focusedCellId()) : new Set<number>();
  });
  readonly sameNumberCellIds = computed(() => {
    const board = this._board();
    return board ? computeSameNumberCellIds(board, this._focusedCellId()) : new Set<number>();
  });

  readonly hasUserEntries = computed(() => {
    const board = this._board();
    if (!board) {
      return false;
    }
    return board.cells.some((c) => !c.isPrefilled && c.userValue !== 0);
  });
  readonly canValidate = computed(() => this._status() === 'playing' && this.hasUserEntries());
  readonly canSolve = computed(() => this._status() === 'playing' && this._board() !== null);

  constructor() {
    this.wireValidateAction();
    this.wireSolveAction();
    this.wirePersistenceEffect();
  }

  startGame(difficulty: GameDifficultyLevel): void {
    this.storage.clearSnapshot();
    this._status.set('loading');
    this._difficulty.set(difficulty);
    this._resolvedDifficulty.set(difficulty === 'random' ? null : difficulty);
    this._validationResult.set(null);
    this._solveResult.set(null);
    this._error.set(null);
    this._focusedCellId.set(null);
    this.api
      .getBoard(difficulty)
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe({
        next: (response) => {
          this._board.set(this.mapper.toUiBoard(response.board));
          this._status.set('playing');
        },
        error: (error: HttpErrorResponse) => {
          this._error.set(this.errors.handleHttpError(error));
          this._status.set('error');
        },
      });
  }

  enterDigit(cellId: number, digit: number): void {
    const board = this._board();
    if (!board) {
      return;
    }
    this._board.set(this.mapper.updateUserValue(board, cellId, digit));
  }

  clearCell(cellId: number): void {
    this.enterDigit(cellId, 0);
  }

  focusCell(cellId: number | null): void {
    this._focusedCellId.set(cellId);
  }

  validate(): void {
    if (!this.canValidate()) {
      return;
    }
    this.validateAction.next();
  }

  solve(): void {
    if (!this.canSolve()) {
      return;
    }
    this.solveAction.next();
  }

  restoreFromStorage(): void {
    const snapshot = this.storage.loadSnapshot();
    if (!snapshot) {
      return;
    }
    const original = this.mapper.toUiBoard(snapshot.originalBoard);
    const restored: SudokuBoard = {
      cells: original.cells.map((cell) => {
        const userValue = snapshot.currentBoard[cell.row]?.[cell.column] ?? 0;
        return { ...cell, userValue };
      }),
    };
    this._board.set(restored);
    this._difficulty.set(snapshot.difficulty);
    this._resolvedDifficulty.set(snapshot.difficulty === 'random' ? null : snapshot.difficulty);
    this._status.set('playing');
    this.snackBar.open(APP_CONFIG.snackbar.messages.gameRestored, 'Dismiss', {
      duration: APP_CONFIG.snackbar.durationMs,
    });
  }

  reset(): void {
    this._status.set('idle');
    this._difficulty.set(null);
    this._resolvedDifficulty.set(null);
    this._board.set(null);
    this._focusedCellId.set(null);
    this._validationResult.set(null);
    this._solveResult.set(null);
    this._error.set(null);
    this.storage.clearSnapshot();
  }

  private wireValidateAction(): void {
    this.validateAction
      .pipe(
        tap(() => this._status.set('loading')),
        switchMap(() => {
          const board = this._board();
          if (!board) {
            return EMPTY;
          }
          return this.api.validateBoard(this.mapper.toWire(board)).pipe(
            catchError((error: HttpErrorResponse) => {
              this._error.set(this.errors.handleHttpError(error));
              this._status.set('error');
              return EMPTY;
            }),
          );
        }),
        takeUntilDestroyed(this.destroyRef),
      )
      .subscribe((response) => {
        this._validationResult.set(response.status);
        this._status.set(response.status === 'solved' ? 'finished' : 'playing');
        this.snackBar.open(
          response.status === 'solved'
            ? APP_CONFIG.snackbar.messages.validateSolved
            : APP_CONFIG.snackbar.messages.validateBroken,
          'Dismiss',
          { duration: APP_CONFIG.snackbar.durationMs },
        );
      });
  }

  private wireSolveAction(): void {
    this.solveAction
      .pipe(
        tap(() => this._status.set('loading')),
        switchMap(() => {
          const board = this._board();
          if (!board) {
            return EMPTY;
          }
          return this.api.solveBoard(this.mapper.toWire(board)).pipe(
            catchError((error: HttpErrorResponse) => {
              this._error.set(this.errors.handleHttpError(error));
              this._status.set('error');
              return EMPTY;
            }),
          );
        }),
        takeUntilDestroyed(this.destroyRef),
      )
      .subscribe((response) => {
        this._solveResult.set(response.status);
        if (response.status === 'unsolvable') {
          this._status.set('playing');
          this.snackBar.open(APP_CONFIG.snackbar.messages.solveUnsolvable, 'Dismiss', {
            duration: APP_CONFIG.snackbar.durationMs,
          });
          return;
        }
        const board = this._board();
        if (board) {
          this._board.set(this.mapper.applySolution(board, response.solution));
        }
        this._resolvedDifficulty.set(response.difficulty);
        this._status.set('finished');
        this.snackBar.open(
          response.status === 'solved'
            ? APP_CONFIG.snackbar.messages.solveSolved
            : APP_CONFIG.snackbar.messages.solveBroken,
          'Dismiss',
          { duration: APP_CONFIG.snackbar.durationMs },
        );
      });
  }

  private wirePersistenceEffect(): void {
    let debounceHandle: ReturnType<typeof setTimeout> | null = null;
    effect(() => {
      const board = this._board();
      const difficulty = this._difficulty();
      if (!board || !difficulty) {
        return;
      }
      if (debounceHandle) {
        clearTimeout(debounceHandle);
      }
      const snapshot: GameSnapshot = {
        schemaVersion: 1,
        difficulty,
        originalBoard: this.snapshotOriginal(board),
        currentBoard: this.mapper.toWire(board),
      };
      debounceHandle = setTimeout(() => {
        this.storage.saveSnapshot(snapshot);
      }, 250);
    });
  }

  private snapshotOriginal(board: SudokuBoard): readonly (readonly number[])[] {
    const rows: number[][] = Array.from({ length: 9 }, () => Array<number>(9).fill(0));
    for (const cell of board.cells) {
      const targetRow = rows[cell.row];
      if (targetRow) {
        targetRow[cell.column] = cell.originalValue;
      }
    }
    return rows;
  }
}

import { Directive, HostListener } from '@angular/core';

const ALLOWED_NON_DIGITS = new Set([
  'Backspace',
  'Delete',
  'Tab',
  'Escape',
  'Enter',
  'ArrowUp',
  'ArrowDown',
  'ArrowLeft',
  'ArrowRight',
  'Home',
  'End',
]);

  selector: '[appDigitOnly]',
  standalone: true,
})
export class DigitOnlyDirective {
  @HostListener('keydown', ['$event'])
  onKeydown(event: KeyboardEvent): void {
    if (ALLOWED_NON_DIGITS.has(event.key)) {
      return;
    }
    if (/^[1-9]$/.test(event.key)) {
      return;
    }
    event.preventDefault();
  }
}

import { ChangeDetectionStrategy, Component, input, output } from '@angular/core';

import type { SudokuCell } from '../../../../models/sudoku-board';

@Component({
  selector: 'app-sudoku-cell',
  standalone: true,
  imports: [CoalesceEmptyPipe],
  templateUrl: './sudoku-cell.component.html',
  styleUrl: './sudoku-cell.component.css',
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class SudokuCellComponent {
  readonly cell = input.required<SudokuCell>();
  readonly isFocused = input.required<boolean>();
  readonly isRelatedHighlight = input.required<boolean>();
  readonly isSameNumberHighlight = input.required<boolean>();
  readonly isConflict = input.required<boolean>();
  readonly isSolverFilled = input.required<boolean>();

  readonly digitEntered = output<number>();
  readonly cellCleared = output<void>();
  readonly focusRequested = output<void>();

  onKeydown(event: KeyboardEvent): void {
    if (event.key === 'Escape') {
      (event.currentTarget as HTMLElement | null)?.blur();
      event.preventDefault();
      return;
    }
    if (this.cell().isPrefilled) {
      return;
    }
    if (/^[1-9]$/.test(event.key)) {
      this.digitEntered.emit(Number(event.key));
      event.preventDefault();
      return;
    }
    if (['Backspace', 'Delete', '0'].includes(event.key)) {
      this.cellCleared.emit();
      event.preventDefault();
    }
  }

  onFocus(): void {
    this.focusRequested.emit();
  }

  focusHost(host: HTMLElement): void {
    host.focus();
  }
}

// sudoku-board.component.ts
import {
  ChangeDetectionStrategy,
  Component,
  computed,
  ElementRef,
  HostListener,
  input,
  output,
  viewChildren,
} from '@angular/core';
import type { SudokuBoard } from '../../../../models/sudoku-board';
import { SudokuCellComponent } from '../sudoku-cell/sudoku-cell.component';

const SIZE = 9;
const ARROW_KEYS = new Set(['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight']);

@Component({
  selector: 'app-sudoku-board',
  standalone: true,
  imports: [SudokuCellComponent],
  templateUrl: './sudoku-board.component.html',
  styleUrl: './sudoku-board.component.css',
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class SudokuBoardComponent {
  readonly board = input.required<SudokuBoard>();
  readonly focusedCellId = input.required<number | null>();
  readonly relatedCellIds = input.required<ReadonlySet<number>>();
  readonly sameNumberCellIds = input.required<ReadonlySet<number>>();
  readonly conflictCellIds = input.required<ReadonlySet<number>>();

  readonly digitEntered = output<{ cellId: number; digit: number }>();
  readonly cellCleared = output<number>();
  readonly cellFocused = output<number>();

  private readonly cellElements = viewChildren(SudokuCellComponent, {
    read: ElementRef<HTMLElement>,
  });

  readonly rows = computed(() => {
    const cells = this.board().cells;
    const grouped: Array<typeof cells> = [];
    for (let r = 0; r < SIZE; r++) {
      grouped.push(cells.slice(r * SIZE, (r + 1) * SIZE));
    }
    return grouped;
  });

  onDigit(cellId: number, digit: number): void {
    this.digitEntered.emit({ cellId, digit });
  }

  onClear(cellId: number): void {
    this.cellCleared.emit(cellId);
  }

  onFocus(cellId: number): void {
    this.cellFocused.emit(cellId);
  }

  @HostListener('keydown', ['$event'])
  onArrowKey(event: KeyboardEvent): void {
    if (!ARROW_KEYS.has(event.key)) {
      return;
    }
    const currentId = this.focusedCellId();
    if (currentId === null) {
      return;
    }
    const row = Math.floor(currentId / SIZE);
    const column = currentId % SIZE;
    let nextRow = row;
    let nextColumn = column;
    if (event.key === 'ArrowUp') nextRow = Math.max(0, row - 1);
    if (event.key === 'ArrowDown') nextRow = Math.min(SIZE - 1, row + 1);
    if (event.key === 'ArrowLeft') nextColumn = Math.max(0, column - 1);
    if (event.key === 'ArrowRight') nextColumn = Math.min(SIZE - 1, column + 1);
    const nextId = nextRow * SIZE + nextColumn;
    if (nextId === currentId) {
      return;
    }
    event.preventDefault();
    this.cellFocused.emit(nextId);
    const targetRef = this.cellElements()[nextId];
    const target = targetRef?.nativeElement.querySelector<HTMLElement>('[role="gridcell"]');
    target?.focus();
  }
}
